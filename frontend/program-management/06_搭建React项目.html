const resolve = (dir) => path.resolve(__dirname, dir)

	webpack: {

5. husky + lint-staged ​

  1. 安装husky
shell
npm install husky -D
1
  1. 然后执行脚本: 这个命令会在package.json的scripts字段里加一条 { "prepare": "husky install" }
shell
npm set-script prepare "husky install"
1
  1. 安装lint-staged
shell
npm install lint-staged -D
1
  1. 配置package.json
json
"scripts": {
	"start": "craco start",
	"build": "craco build",
	"test": "craco test",
	"prettier": "prettier --write .",
	"prepare": "husky install",
	"lint": "lint-staged",
},
"lint-staged": {
	"*.{js,jsx}": [
		"eslint --fix",
		"git add"
	]
},

6. commitlint ​

  1. 安装代码校验依赖

npm i commitlint @commitlint/config-conventional -D
1
  1. 初始化 husky
shell

1
  1. 跟目录下创建commitlint.config.js
javascript

	extends: ['@commitlint/config-conventional'], // 检测规则
	rules: {}
}

json

	"source.fixAll": true

	"javascript",
	"javascriptreact",
	"typescript",
